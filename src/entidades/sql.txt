CREATE TABLE Emprestimo (IDEmprestimo INTEGER PRIMARY KEY AUTO_INCREMENT, IDUsuario INTEGER NOT NULL, IDLivro INTEGER NOT NULL, DataEmprestimo DATE NOT NULL, DataDevolucao DATE, Prazo DATE NOT NULL, TempoDeEmprestimo INTEGER NOT NULL, Observacoes VARCHAR(400), FOREIGN KEY(IDUsuario) REFERENCES Usuario (IDUsuario), FOREIGN KEY(IDLivro) REFERENCES Livro (IDLivro));
CREATE TABLE Livro (IDLivro INTEGER PRIMARY KEY AUTO_INCREMENT, Titulo VARCHAR(100) NOT NULL, Capa VARCHAR(300), ISBN VARCHAR(13) NOT NULL, AUTORES VARCHAR(300) NOT NULL, QTDEXEMPLARES INTEGER NOT NULL, QTDEMPRESTADOS INTEGER NOT NULL);
CREATE TABLE Usuario (IDUsuario INTEGER PRIMARY KEY AUTO_INCREMENT, Nome VARCHAR(150) NOT NULL, NumDoc VARCHAR(20) NOT NULL, TipoDoc INTEGER NOT NULL, Telefone VARCHAR(11), Status INTEGER NOT NULL, Email VARCHAR(100), Motivo INTEGER, Foto VARCHAR(300), Cidade VARCHAR(50), UF VARCHAR(2), CEP VARCHAR(8), Rua VARCHAR(50), Bairro VARCHAR(50), Numero VARCHAR(4));
