CREATE TABLE Endereco (IDEndereco INTEGER PRIMARY KEY AUTO_INCREMENT, Cidade VARCHAR(50) NOT NULL, UF VARCHAR(2) NOT NULL, CEP VARCHAR(8) NOT NULL, Rua VARCHAR(50) NOT NULL, Bairro VARCHAR(50) NOT NULL);
CREATE TABLE Livro (IDLivro INTEGER PRIMARY KEY AUTO_INCREMENT, Titulo VARCHAR(100) NOT NULL, Capa VARCHAR(300), ISBN VARCHAR(13) NOT NULL, AUTORES VARCHAR(300) NOT NULL, QTDEXEMPLARES INTEGER NOT NULL, QTDEMPRESTADOS INTEGER NOT NULL);
CREATE TABLE Usuario (IDUsuario INTEGER PRIMARY KEY AUTO_INCREMENT, Nome VARCHAR(150) NOT NULL, NumDoc VARCHAR(20) NOT NULL, TipoDoc VARCHAR(22) NOT NULL, Telefone VARCHAR(11), Status INTEGER, Email VARCHAR(100), Motivo VARCHAR(100), Foto VARCHAR(300), IDEndereco INTEGER, FOREIGN KEY(IDEndereco) REFERENCES ENDERECO (IDEndereco));
CREATE TABLE Emprestimo (IDEmprestimo INTEGER PRIMARY KEY AUTO_INCREMENT, IDUsuario INTEGER NOT NULL, IDLivro INTEGER NOT NULL, DataEmprestimo DATE NOT NULL, DataDevolucao DATE, TempoDeEmprestimo INTEGER NOT NULL, Observacoes VARCHAR(400), FOREIGN KEY(IDUsuario) REFERENCES Usuario (IDUsuario), FOREIGN KEY(IDLivro) REFERENCES Livro (IDLivro));
